SELECT 
         [activityid]							as ActivityID
		,[subject]								as Subject
        ,[gfps_accountid]						as AccountID
	    ,[gfps_appointment_typecode]			as AppointmentTypecode
		,[gfps_appointment_typecodename]		as AppointmentTypecodename
        ,[gfps_bi_appointment_typecode]			as BiAppointmentTypecode
	    ,[gfps_bi_appointment_typecodename]		as BiAppointmentTypecodename
		,[ownerid]								as OwnderID
		,[owneridname]							as Owner
		,[scheduledend]							as Scheduledend
		,MONTH(GETDATE())
  FROM [dbo].[crm_Appointments]
  where ownerid in  ('0ba12ab6-8c09-ed11-82e5-6045bd8b970e','063c470c-f2cf-e911-a82e-000d3a3a6eb0') 
  and MONTH([scheduledend]) = MONTH(GETDATE()) AND YEAR([scheduledend]) = YEAR(GETDATE())
  and [gfps_bi_appointment_typecode] = 765740002
